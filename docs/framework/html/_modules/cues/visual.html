[docs]    def createWindow(self, windowConfig):
        p1 = ctypes.c_uint32(windowConfig.width)
        p2 = ctypes.c_uint32(windowConfig.height)
        p3 = ctypes.c_int32(windowConfig.selectedMonitor)
        bShaderPath = windowConfig.shaderPath.encode('utf_8')
        p4 = ctypes.c_char_p(bShaderPath)
        bFontPath = windowConfig.fontPath.encode('utf_8')
        p5 = ctypes.c_char_p(bFontPath)
        
        self.wrappedCreateWindow(ctypes.byref(p1), ctypes.byref(p2), ctypes.byref(p3), p4, p5);
        if (windowConfig.enableVSync):
            self.wrappedEnableVSync()
        else:
            self.wrappedDisableVSync()
        
        self.isActive = 1
