"""This module contains classes resembling acoustic and visual stimuli

It contains acoustic stimuli, represented by \
:class:`~.experiment.stimulus.Sound` and visual stimuli \
represented by the general :class:`~.experiment.stimulus.Frame`

Visual stimuli
--------------

Are grouped into whether they are a

- static :class:`~.experiment.stimulus.Frame`, e.g. \
:class:`~.experiment.stimulus.Mural`, :class:`~.experiment.stimulus.Cross` \
or :class:`~.experiment.stimulus.Rectangle`
- :class:`~.experiment.stimulus.Dynamic` e.g. \
:class:`~.experiment.stimulus.Ball` or :class:`~.experiment.stimulus.Bar`

-------------

[docs]class Sound(_acoustic.AcousticConfig):
    """ Representing an Sound to be played with Speaker
        Example:


        create a "silent" Sound object

        >>> NullSound       = experiment.stimulus.Sound()

        Create a Sound object based on wavfile.wav

        >>> WavFilePath     = "wavfile.wav"
        >>> Sound           = experiment.stimulus.Sound(Speaker, WavFilePath)

        see also

        Speaker
    """
    LoadedBefore = False

    def __init__(self, Speaker=_NullDevice(), filePath=None,
                 Duration='Automatic'):
        self.Speaker = Speaker
        if Speaker.libActive is False:
            Speaker.open()
        self.soundID = None
        self.Duration = None
        self.filePath = filePath
        if Duration == 'Automatic':
            self.Duration = self.readDuration(filePath)
        else:
            self.Duration = Duration

        self._preload()


[docs]    def readDuration(self, filePath):
        if filePath is not None:
            w = wave.open(self.filePath)
            frames = w.getnframes()
            rate = w.getframerate()
            Duration = frames / float(rate)
            w.close()
            del(w)
        else:
            Duration = 0
        return Duration


    def _preload(self):
        if self.Speaker.libActive is False:
            pass
        else:
            if self.LoadedBefore is False:
                self.soundID = self.Speaker._lib.preLoadFile(self.filePath,
                                                             self.Duration)
                if self.soundID == -1:
                    print("File Access Error, restarting Speaker")
                    self.Speaker.close()
                    self.Speaker.open()
                    self.soundID = self.Speaker._lib.preLoadFile(self.filePath,
                                                                 self.Duration)

                self.LoadedBefore = True
            else:
                raise EnvironmentError('Do not rewriting sound ID' +
                                       'Solution: create new instance')


[docs]    def close(self):
        pass


    def __str__(self):
        return f'Sound from {self.filePath}'

    def __del__(self):
        self.close()
        print("Shutting Down: " + self.__repr__())

[docs]class Rectangle(Frame):
    """
    Example ::

       rectangle = experiment.stimulus.Rectangle(Screen, xPos = 0, yPos = 0,
                                              Height = 100, Width = 100,
                                              Color = "gray")
       rectangle.showNow()

    args
    ----
    Screen: :class:`~.experiment.device.Screen`
        A display Screen is a necessary requirement
    xPos: int
        x-coordinates of center of rectangle in pixes
    yPos: int
        y-coordinates of center of rectangle in pixes
    Height: int
        heigth of rectangle in pixels
    Width: int
        width of rectangle in pixels
    Color: tuple or str
        either a string from :attr:`~.experiment.device.Screen.colors` or \
        a tuple with len three defining the color in RBG, scaled either between
         (0, .5, 1) or (0, 127, 255)

    """
    def __init__(self, Screen, xPos=0, yPos=0, Height=100, Width=100,
                 Color="gray"):
        self._init_Window(Screen)
        self._init_Rect(xPos, yPos, Height, Width, Color)

    def _init_Rect(self, xPos, yPos, Height, Width, Color):
        self._Rect = _visual.VisualConfig()
        self._Rect.height = Height
        self._Rect.width = Width
        self._Rect.xPos = xPos
        self._Rect.yPos = yPos
        self.setColor(Color)


[docs]    def setColor(self, Color="gray"):
        if type(Color) == str:
            self._Rect.ColorRGB = self._Screen.colors[Color]
        elif type(Color) == list and len(Color) == 3:
            if max(Color) > 1:
                self._Rect.ColorRGB = [c / 255 for c in Color]
            else:
                self._Rect.ColorRGB = Color
        else:
            raise Exception('Color not correctly specified\n' +
                            'Solution: Choose a correct color')


    def _draw(self):
        self._Screen._Win.drawRectangle(self._Rect)

[docs]    def setZoom(self, ZoomFactor=1):
        '''Sets the Zoomfactor

        .. admonition:: Sanity Checks

           will use the default of 1 if a 0 is used as Zoomfactor
        '''
        self._ZoomFactor = (ZoomFactor, 1)[ZoomFactor == 0]
        self._init_Cross()


    def _init_Cross(self):
        self._Cross = _visual.VisualConfig()
        self._Cross.ColorRGB = self._Screen.colors["white"]
        self._Cross.height = int(self._Screen.diag * (self._ZoomFactor / 10))
        self._Cross.width = self._Cross.height
        self._Cross.thickness = int(self._Cross.height * 0.2)
        self._Cross.xPos = 0
        self._Cross.yPos = 0

    def _draw(self):
        self._Screen._Win.drawFixationCross(self._Cross)

[docs]class Dynamic(Frame):
    '''Base class resembling dynamic visual stimuli

    args
    ----
    Screen: :class:`~.experiment.device.Screen`
        A display Screen is a necessary requirement
    Target: float
        value between 0 and 100 defining the target level
    State: float
        value between 0 and 100 defining the current level
    Tolerance: float
        value between 0 and 100 defining the range of values \
        accepted around target as sufficient


    .. caution::
       as this is the base class, most methods are abstract and need to be
       implemented, e.g. as in :class:`~.experiment.stimulus.Bar` or
       :class:`~.experiment.stimulus.Ball`

    .. tip::
        - you can use print(Dynamic) for debugging purposes
        - will use the default colors of its \
        :class:`~.experiment.device.Screen` change these accordingly
    '''

        self.setTolerance(Tolerance)
        self._init_BackGrnd()
        self._init_State()
        self._init_Target()
        self.updateTarget(Target)
        self.updateState(State)

[docs]    def setTolerance(self, Tolerance=10):
        'Sets the tolerance'
        self.Tolerance = Tolerance


    def _updateColor(self, Above, Below, On):
        if self.StateLevel > (self.TargetLevel + self.Tolerance / 2):
            self._State.ColorRGB = Above
            # print("Above")
        elif self.StateLevel < (self.TargetLevel - self.Tolerance / 2):
            self._State.ColorRGB = Below
            # print("Below")
        else:
            self._State.ColorRGB = On
            # print("Equal")

    def _check(self):
        self._updateColor(self._Screen._ABOVE, self._Screen._BELOW,
                          self._Screen._ON)

    def __str__(self):
        coretext = ("State: " + str(self.StateLevel) +
                    " versus Target " + str(self.TargetLevel))
        if self.StateLevel > (self.TargetLevel + self.Tolerance / 2):
            return coretext + " \x1b[5;30;41mAbove\x1b[0m"
        elif self.StateLevel < (self.TargetLevel - self.Tolerance / 2):
            return coretext + " \x1b[5;30;41mBelow\x1b[0m"
        else:
            return coretext + " \x1b[5;30;42mEqual\x1b[0m"

[docs]class Bar(Dynamic):
    """Dynamic Feedback with a bar changing height

    Example::

        screen  = cues.feedback.Screen()
        bar     = cues.feedback.Bar(Screen)
        bar.showNow()

        for i in range(1,101,1):
            bar.updateState(i)
            bar.showNow()

        bar.clear()


    .. seealso::
       :class:`~.experiment.stimulus.Dynamic`

    """

    def _draw(self):
        self._Screen._Win.drawRectangle(self._BackGrnd)
        self._Screen._Win.drawRectangle(self._State)
        self._Screen._Win.drawRectangle(self._Target)

    def _init_Target(self):
        self._Target = _visual.VisualConfig()
        self._Target.ColorRGB = self._Screen.colors["white"]
        self._Target.height = int(self._State.MaxHeight *
                                  (self.Tolerance / 100))
        self._Target.width = int(self._Screen.width / 4)
        self._Target.xPos = 0

    def _init_State(self):
        self._State = _visual.VisualConfig()
        self._State.ColorRGB = self._Screen._BELOW
        self._State.MaxHeight = int(self._Screen.height * .9)
        self._State.height = 0
        self._State.width = int(self._Screen.width / 5)
        self._State.xPos = 0

    def _init_BackGrnd(self):
        self._BackGrnd = _visual.VisualConfig()
        self._BackGrnd.ColorRGB = self._Screen.colors["gray"]
        self._BackGrnd.height = int(self._Screen.height * 1)
        self._BackGrnd.width = int(self._Screen.width / 4.5)
        self._BackGrnd.xPos = 0
        self._BackGrnd.yPos = 0


[docs]    def updateState(self, Level):
        self.StateLevel = Level
        height = (self._State.MaxHeight * Level) / 100
        if isnan(height):
            height = 0
        self._State.height = int(height)
        self._State.yPos = int((self._Screen.height / 2) -
                               (self._State.height / 2))

        self.TargetLevel = Level
        self._Target.yPos = int((self._Screen.height / 2) -
                                (self._State.MaxHeight * Level) / 100)

        self._Target = _visual.VisualConfig()
        self._Target.ColorRGB = self._Screen.colors["light"]
        self._Target.height = int(self._Screen.height *

        self._State = _visual.VisualConfig()
        self._State.ColorRGB = self._Screen.colors["light"]
        # scaling by .5 because it looks otherwise too large
        self._State.radius = int(self._Screen.height *
                                 (self.Tolerance / 100) *
                                 self._ZoomFactor * 0.5)
        self._State.xPos = 0

        self._BackGrnd = _visual.VisualConfig()
        self._BackGrnd.ColorRGB = self._Screen.colors["gray"]
        self._BackGrnd.Tolerance = self.Tolerance * 1.25
        self._BackGrnd.height = int(self._Screen.height *
                                    (self._BackGrnd.Tolerance / 100) *
                                    self._ZoomFactor)
        self._BackGrnd.width = int(self._Screen.width)
        self._BackGrnd.xPos = 0
        self._BackGrnd.yPos = 0
