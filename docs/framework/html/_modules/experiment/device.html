[docs]    def buffer(self, Sound):
        '''Puts a sound object into the stream buffer

        args
        ----
        sound: :class:`~.experiment.protocol.Sound`
            A sound to played


        .. admonition:: sideeffect

            flushes the current buffer
        '''
        if self._ThreadState == 'Buffered':
            self.flush()
            self.buffer(Sound)

        elif self._ThreadState == 'Empty':
            self._lib.setupFileDirectly(Sound.soundID)
            self._thread = _thread(target=self._lib.playFileNow, args=())
            self._ThreadState = 'Buffered'
            self._sound = Sound

        else:
            pass



[docs]    def flush(self):
        '''Flushes the stream buffer
        '''
        if self._ThreadState == 'Running':
            self._thread.join()
            del(self._thread)
            self._ThreadState = 'Empty'
        elif self._ThreadState == 'Buffered':
            del(self._thread)
            self._ThreadState = 'Empty'
        else:
            pass



[docs]    def play(self):
        '''Plays the sound object currently in the stream buffer

        notes
        -----
        will ignore the command if it is still playing the last sound

        '''
        if self._ThreadState == 'Buffered':
            self._thread.start()
            self._ThreadState = 'Running'
        else:
            pass

[docs]    def replay(self):
        '''Replays the sound object played last if still in history or buffer
        '''

        if self._sound is not None:
            if self._ThreadState == 'Empty':
                self.buffer(self._sound)
            elif self._ThreadState == 'Buffered':
                pass
            elif self._ThreadState == 'Running':
                self.flush()
                self.buffer(self._sound)
            self.play()
        else:
            raise LookupError('No sound in history')

[docs]    def scan_workspace(self, workingFilePath='C:\Vision\Raw Files'):
        '''
            Records 200ms of data with the current workspace into
            workingFilePath and reads channels and Fs from this recording
        '''

        if _os.path.exists(workingFilePath) is False:
            raise Exception('workingFilePath not found')

        # switch to view
        if not self.state() == 1:
            self.bvr.Acquisition.ViewData()
            nanoClock().sleep(5)
        else:
            nanoClock().sleep(.1)

        # save 100 ms of data
        self.bvr.Acquisition.StartRecording(_os.path.join(
            workingFilePath, "tmp.eeg"))
        nanoClock().sleep(.1)
        self.bvr.Acquisition.StopRecording()
        self.bvr.Acquisition.StopViewing()  # prevents file lock by bvr

        # view and save 200 ms of data
        fname = _os.path.join(workingFilePath, "tmp.vhdr")
        timeOut = 0
        while _os.path.exists(fname) is False:
            nanoClock().sleep(.1)
            timeOut += 1
            if timeOut > 30:  # if it takes longer than 3 seconds
                raise TimeoutError('Could not find the tempory datafile')
                break

        # retrieve chan and fs
        conf_recording = _io.load_brain_vision_data(fname)
        self.chan = conf_recording.axes[1]
        self.fs = conf_recording.fs

        # delete temporary files
        for extension in ('vmrk', 'eeg', 'vhdr'):
            _os.remove(_os.path.join(workingFilePath, 'tmp.' + extension))
        self.bvr.Acquisition.ViewData()
        # return fs and chan
        return self.fs, self.chan

        """
        if hasattr(self, "_arduino"):

            if PortNumber < 10:
                try:
                    arduino = _serial.Serial('COM' + str(PortNumber),
                                             BaudRate, timeout=.1)
                    return 'COM' + str(PortNumber), arduino

                                                       PortNumber + 1)
                    return port, arduino
            else:

                                       'at a ComPort below COM10')

            attempt_connection(self._BaudRate,
                               int(self._ComPort.split('COM')[1]))

        self.flush = self._arduino.flush

    def _trigger_mf255(self, stimKey: str, settleDuration=2):
        """trigger in mf255 mode

        Send the arg:StimKey to the SerialTrigger
        with default settle duration of 100 ms

        only 7-Bit Messages are allowed,
        i.e. Stimkey mapping to values of 0-127
        Written on D0-D6, then D7 is flipped
        """
        triggerNumber = self.stimDict[stimKey]
        if triggerNumber <= 127:
            sendHIGH = triggerNumber | 128
            sendLOW = triggerNumber | 0
            highMessage = sendHIGH.to_bytes(1, 'big')
            lowMessage = sendLOW.to_bytes(1, 'big')
            resetMessage = (0).to_bytes(1, 'big')
        else:
            raise ConnectionError('Only 7-Bit Messages are allowed')

        self.interface.write(lowMessage)
        self.sleep(0.001)  # at least 20 ns
        self.interface.write(highMessage)
        self.sleep(settleDuration / 1000)  # milliseconds scaling
        self.interface.write(resetMessage)

    def _stimkey_generator(self):
        def pd():
            newPermutation = list(_np.random.permutation(
                                  list(self.stimDict.keys())))
            keyReservoir = newPermutation
            for cnt, key in enumerate(keyReservoir):
                yield key
                if cnt == len(keyReservoir) - 1:
                    newPermutation = list(_np.random.permutation(
                                          list(self.stimDict.keys())))
                    keyReservoir.extend(newPermutation)
        permDict = pd()
        return permDict.__next__
