[docs]def register_triggerdevice(td=None):
    '''registers a TriggerDevice

        returns
        -------
        function: a function for this device


        notes
        -----
        the returned function in turn registers marker values and
        returns a function that sends this marker


        Example::

            m = register_triggerdevice(None)(None)
            Marker = register_triggerdevice(HardWareTrigger)
            m   = Marker(5) # creates a function to send 5 with this device
            m.get() # returns the value of the device

        .. tip::
            `m = register_triggerdevice(None)(None)` sends to a nulldevice

        .. seealso::
            :class:`~.experiment.protocol.Cue`

    '''
    def register_markervalue(mv=None):
        def get_mv():
            return str(mv)

        if mv == None:
            def send_marker():
                pass
        else:
            def send_marker():
                td.send(mv)

        send_marker.__str__ = get_mv
        return send_marker

    return register_markervalue

[docs]class Cue():
    """Class representing a Cue

    args
    ----
    Sound: :class:`~.experiment.stimulus.Sound`
        The acoustic part of the cue
    Frame: :class:`~.experiment.stimulus.Frame`
        The visual part of the cue
    Duration: float
        Duration in seconds
    Marker: function
        a function sending a trigger on evaluation


    Example::

        speaker     = device.Speaker()
        sound       = stimulus.Sound(speaker, pathToWavFile)
        screen      = device.Screen()
        frame       = stimulus.Cross(screen)
        duration    = 5
        interface   = device.SerialTrigger()
        marker      = protocol.register_triggerdevice(interface)
        cue         = protocol.Cue(sound, frame, duration, marker(6))
        interface.connect()
        cue.present()
        interface.close()


    .. tip::
        use the dictionary of cues, sounds and visuals in \
        :mod:`~.experiment.api`

    .. seealso::
        - :func:`~.experiment.protocol.register_triggerdevice`
        - :mod:`~.experiment.stimulus`
    """

    def __init__(self, Sound, Frame, Duration,
                 trigger=register_triggerdevice(None)(None)):
        self.Sound = Sound
        self.Frame = Frame
        self.Timing = Timing(Duration)
        self.Trigger = trigger

        if isinstance(self.Frame, experiment.stimulus.Dynamic):
            self.update = self._dynamic_update
            if Sound is None:
                self.present = self._dynamic_silent
            else:
                self.present = self._dynamic_loud
        else:
            if Sound is None:
                self.present = self._static_silent
            else:
                self.present = self._static_loud

    def _static_silent(self):
        self.Timing.ready()
        self.Frame.showNow()
        self.Trigger()
        self.Frame.showDelayed(self.Timing)
        self.Sound.Speaker.flush()

    def _static_loud(self):
        self.Sound.Speaker.buffer(self.Sound)
        self.Timing.ready()
        self.Frame.showNow()
        self.Trigger()
        self.Sound.Speaker.play()
        self.Frame.showDelayed(self.Timing)
        self.Sound.Speaker.flush()

    def _dynamic_loud(self):
        self.Sound.Speaker.buffer(self.Sound)
        self.Trigger()
        self.Frame.showNow()
        self.Sound.Speaker.play()
        self.Sound.Speaker.flush()

    def _dynamic_silent(self):
        self.Frame.showNow()
        self.Trigger()
        self.Sound.Speaker.flush()

    def _static_update(self, Ignore):
        pass

    def _dynamic_update(self, Percent):
        self.Frame.updateState(Percent)
        self.Frame.showNow()

    def __str__(self):
        m = self.Trigger.__str__()
        s = self.Sound.__str__()
        v = self.Frame.__str__()
        return f'{v} with {s} marked by {m}'

    def __del__(self):
        pass

[docs]def Logger(filepath=r'C:\recording', fname="session.log"):
    global logger
    logger = _logging.getLogger(__name__)
    logger.setLevel(_logging.INFO)

    ch = _logging.StreamHandler()
    ch.setLevel(_logging.INFO)
    ch.setFormatter(_logging.Formatter('%(filename)s - %(message)s'))
    logger.addHandler(ch)

    fh = _logging.FileHandler(_os.path.join(filepath, fname))
    fh.setLevel(_logging.INFO)
    fh.setFormatter(
        _logging.Formatter('%(asctime)s - %(filename)s - %(message)s'))
    logger.addHandler(fh)
    return logger

[docs]    def dump(self):
        self._dump_config()
        if not hasattr(self, 'name'):
            raise FileNotFoundError('''No name was set for the subject\n
                                    Solution: Set a name first''')

        mainfolder = _os.path.dirname(self.folder)
        recfolder = _os.path.join(mainfolder, 'sessions')
        dst = _os.path.join(recfolder, _now("%Y_%m_%d") +
                            '_' + self.name)
        if not _os.path.exists(recfolder):
            _os.mkdir(recfolder)
        _shutil.move(self.folder, dst)

[docs]    @staticmethod
    def filename(folderPath, fileName, ext='.eeg'):
        '''
            Takes a folder and filename and extension as argument,
            if file already exists, it increases its indicator value and
            returns the new filename
        '''
        run = 0
        return_fname = Run.join(folderPath, run, fileName, ext)
        while _os.path.exists(return_fname):
            run = run + 1
            return_fname = Run.join(folderPath, run, fileName, ext)
        return return_fname



[docs]    @staticmethod
    def FileIterator(fullPath):
        run = 0
        filename, ext = _os.path.splitext(fullPath)
        folderPath, fileName = _os.path.split(filename)
        _, fileName = fileName.split('_', 1)

        return_fname = Run.join(folderPath, run, fileName, ext)
        while _os.path.exists(return_fname):
            yield return_fname, run
            run += 1
            return_fname = Run.join(folderPath, run, fileName, ext)
