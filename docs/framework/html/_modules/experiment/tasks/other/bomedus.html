"""

[docs]def start(Duration = 600,
          qualifier = '01',
          sham = False,
          threshold = 4,
          saveDir = r'C:\projects\bomedus\recording'
          ):
    import experiment.api as xp
 
    # Load Cues from base dictionary           
    Cue_Pre    = xp.Cue(xp.SOUND['null'],  xp.STATIC['pre'], 1)
    Cue_Post   = xp.Cue(xp.SOUND['hint'], xp.STATIC['post'], 5)    
    Cue_Bomedus  = xp.Cue(xp.SOUND['peep'], xp.STATIC['cross'], Duration) 
   

    # %% START RECORDING 
    eegFileName = xp.Run.filename(saveDir,'bomedus_' + qualifier,'.eeg')       
    xp.Screen.open()
    with xp.device.VisionRecorder() as bvr:        
        with xp.device.ArduinoTrigger() as interface:
            interface.set_threshold(threshold)
            interface.enquire()
            bvr.start_recording(eegFileName)          
        # %% EXPERIMENTAL BLOCK                      
            Cue_Pre.present() #Run begins        
            if not sham:
                interface.trigger_loud(1)
            Cue_Bomedus.present() #peep and show cross   
            if not sham:
                interface.trigger_loud(0)
            Cue_Post.present() #final peep
        
    # %% STOP RECORDING         
    xp.Screen.close()

    
# %%
if __name__ == '__main__':
    
    from multiprocessing import freeze_support
    import sys
    sys.path.append(r"C:\py3tools\StimulationFramework_nightly\code")     
     
    freeze_support()
    start()
