class _psd:
    @staticmethod
    def yule(x, order = 250, nfft = 1000, Fs = 1000):
        c       = np.correlate(x, x, 'full')
        mid     = len(c)//2
        acov    = c[mid:]
        ac      = acov/acov[0]
        order   = np.min([order, len(ac)])
        R       = scipy.linalg.toeplitz(ac[:order])
        r       = ac[1:order+1]
        ywPhi   = scipy.linalg.inv(R).dot(r)
        a       = np.concatenate([np.ones(1), -ywPhi])
        w, h    = scipy.signal.freqz(1, a)
        db      = 10*np.log10(2*(np.abs(h)/len(h)))
        f       = w/np.pi * Fs/2
        return f, db

    @staticmethod
    def welch(x, order = 250, nfft = 1000, Fs = 1000):
        f, h 	= scipy.signal.welch(x, nperseg = order, nfft = nfft, fs = Fs)
        db 		= 10*np.log10(2*(np.abs(h)/len(h)))
        return f, db

    def __init__(self, AmpSource,
                 chanIdx = [1,2,3],
                 freqRange = [16,22],
                filterBand    = [10,30],
                window      = 500,
                updateRate = 50, #in ms
                valBufferLen = 100,
                featureWeights = np.ones((15,1))/15,
                threshold = 0
                ):


        self.parms      = {}
        self.parms['chanIdx']       = chanIdx
        self.parms['chanNames']     = AmpSource.chan[chanIdx]
        self.parms['freqRange']     = freqRange
        self.parms['filterBand']    = filterBand
        self.parms['Fs']            = AmpSource.fs
        self.parms['window']        = window
        self.parms['updateRate'] = updateRate / 1000
        self.parms['featureWeights'] = featureWeights
        self.parms['valBufferLen'] = int(valBufferLen)
        self.parms['threshold'] = threshold

        self.AmpSource = AmpSource
        self._register_timer()
        self._register_filter(filterBand, AmpSource.fs)
        self._register_psd()

        self.featureBuffer  = np.zeros((len(self.parms['freqBins']),len(self.parms['chanIdx'])))
        self.valBuffer      = []

        self.normMU = 0
        self.normSD = 1

    def _register_psd(self):
        psd_order 	= np.floor(self.parms['window']/2)
        nfft    	= self.parms['Fs']
        self.parms['psdConfig'] = {'order':psd_order,
                                  'nfft':nfft,
                                  'Fs':self.parms['Fs']
                                  }
        self.psd = _psd.welch
        fR = np.arange(0,self.parms['Fs']/2,1/nfft)
        fF = self.parms['freqRange']
        self.parms['freqBins']  = [fi for fi,ff in
                  enumerate(fR)
                  if (fi>np.min(fF) and fi<np.max(fF))]

    def _register_filter(self, filterBand, Fs):
        Nyqist  = Fs/2
        b, a    = scipy.signal.butter(2, [filterBand[0] / Nyqist,
                                       filterBand[1]/ Nyqist], btype='band')
        self.filter = functools.partial(scipy.signal.filtfilt,
                                            b, a)

    def _data2feature(self, data):
        if len(np.shape(data)) < 2:
            data = np.atleast_2d(data).T
        data = data[self.parms['window']:,
                               self.parms['chanIdx']
                               ].T

        for chanID,chanData in enumerate(data):
            tmp     = self.filter(chanData.T)
            f, psd  = self.psd(tmp,**self.parms['psdConfig'])
            self.featureBuffer[:,chanID] =  psd[self.parms['freqBins']]
        return self.featureBuffer

    def _feature2val(self,feature):
        # weight the feature
        #self.rawVal = feature.flatten().dot(self.parms['featureWeights'])
        self.rawVal = np.mean(feature)
        return self.rawVal

    def _val2norm(self, rawVal, weightUpdate):
        if not weightUpdate:
            pass
        elif weightUpdate:
            self.valBuffer.append(rawVal)
            if len(self.valBuffer) > self.parms['valBufferLen']:
                self.valBuffer = self.valBuffer[-self.parms['valBufferLen']:]
            self.normSD = np.std(self.valBuffer)
            self.normMU = np.mean(self.valBuffer)
            pass

        self.normOut = (rawVal-self.normMU)/self.normSD
        # addition -> higher thresholds are more difficult to desync
        self.classOut = self.normOut + self.parms['threshold']
        return self.normOut, self.classOut

    def _norm2percent(self, classOut):
        # higher relative power means lower percent value
        self.percent =  (scipy.stats.norm.cdf(classOut))*100
        return self.percent


[docs]    def classify(self, data, weightUpdate = True):
        feature             = self._data2feature(data)
        rawVal              = self._feature2val(feature)
        normOut, classOut   = self._val2norm(rawVal, weightUpdate)
        percent             = self._norm2percent(classOut)
        # values below zero indicate low power -> ERD
        # values above zero indicate: isRest
        isRest               = bool(classOut > 0)
        return isRest, percent



[docs]    def update(self, weightUpdate = True):
        self.timer.reset()
        cnt  = self.timer.delay(self.parms['updateRate'],
                                self.AmpSource.receive_ring)
        if cnt:
            isRest, percent  = self.classify(cnt.data, weightUpdate)
            return isRest, percent
        else:
            return True, 0


    def __str__(self):
        return f'SD= {self.normSD} M= {self.normMU} Val= {self.valBuffer[-1]}'
